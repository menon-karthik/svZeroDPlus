41/**
42 * @brief Windkessel RCR boundary condition.
43 *
44 * Models the mechanical behavior of a Windkessel boundary condition.
45 *
46 * \f[
47 * \begin{circuitikz} \draw
48 * node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
49 * \draw (1,0) node[anchor=south]{$P_{in}$}
50 * to [R, l=$R_p$, *-] (3,0)
51 * to [R, l=$R_d$, *-*] (5,0)
52 * node[anchor=south]{$P_{ref}$}
53 * (3,0) to [C, l=$C$, *-] (3,-1.5)
54 * node[ground]{$P_{C}$};
55 * \end{circuitikz}
56 * \f]
57 *
58 * ### Governing equations
59 *
60 * \f[
61 * R_{d} Q^{e}-P_{c}^{e}+P_{r e f}-R_{d} C \frac{d P_{c}^{e}}{d t}=0
62 * \f]
63 *
64 * \f[
65 * P^{e}-P_{c}^{e}-R_{p} Q^{e}=0
66 * \f]
67 *
68 * ### Local contributions
69 *
70 * \f[
71 * \mathbf{y}^{e}=\left[\begin{array}{lll}P^{e} & Q^{e} &
72 * P_{c}^{e}\end{array}\right]^{T} \f]
73 *
74 * \f[
75 * \mathbf{E}^{e}=\left[\begin{array}{ccc}
76 * 0 & 0 & -R_{d} C \\
77 * 0 & 0 & 0
78 * \end{array}\right]
79 * \f]
80 *
81 * \f[
82 * \mathbf{F}^{e}=\left[\begin{array}{ccc}
83 * 0 & R_{d} & -1 \\
84 * 1 & -R_{p} & -1
85 * \end{array}\right]
86 * \f]
87 *
88 * \f[
89 * \mathbf{c}^{e}=\left[\begin{array}{c}
90 * P_{r e f} \\
91 * 0
92 * \end{array}\right]
93 * \f]
94 *
95 * @tparam T Scalar type (e.g. `float`, `double`)
96 */

99 public:

214template <typename T>
215void WindkesselBC<T>::update_constant(ALGEBRA::SparseSystem<T> &system) {
216 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[0]) = 1.0;
217 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[2]) = -1.0;
218 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[2]) = -1.0;
219}
220
221template <typename T>
222void WindkesselBC<T>::update_time(ALGEBRA::SparseSystem<T> &system, T time) {
223 system.E.coeffRef(this->global_eqn_ids[1], this->global_var_ids[2]) =
224 -params.Rd * params.C;
225 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[1]) =
226 -params.Rp;
227 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[1]) =
228 params.Rd;
229 system.C(this->global_eqn_ids[1]) = params.Pd;
230}
