42/**
43 * @brief BloodVesselJunction
44 *
45 * Models a junction with one inlet and arbitrary outlets using
46 * MODEL::BloodVessel elements between each inlet and outlet pair.
47 *
48 * \f[
49 * \begin{circuitikz}
50 * \draw node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
51 * \draw (1,0.1) node[anchor=south]{$P_{in}$};
52 * \draw (1,0) to [short, *-] (2.5,0.75);
53 * \draw (1,0) to [short, *-] (2.5,-0.75);
54 * \draw (2.5,0.75) node[anchor=south]{} to [generic, l_=$BV_{1}$, -*]
55 * (4.5,0.75); \draw (2.4,0.75) node[anchor=south]{$Q_{in,1}$}; \draw (4.6,0.75)
56 * node[anchor=south] {$P_{out,1}$}; \draw (2.5,-0.75) node[anchor=south]{} to
57 * [generic, l^=$BV_{2}$, -*] (4.5,-0.75); \draw (2.4,-0.75)
58 * node[anchor=north]{$Q_{in,2}$}; \draw (4.6,-0.75) node[anchor=north]
59 * {$P_{out,2}$}; \draw [-latex] (4.7,0.75) -- (5.5,0.75) node[right]
60 * {$Q_{out,1}$}; \draw [-latex] (4.7,-0.75) -- (5.5,-0.75) node[right]
61 * {$Q_{out,2}$}; \end{circuitikz} \f]
62 *
63 * ### Governing equations
64 *
65 * The governing equations are mainly defined by the blood vessel elements
66 * (see MODEL::BloodVessel for more information). For the sake of brevity,
67 * the blood vessel contributions are not repeated here. In addition to the
68 * individual blood vessel contributions, the following equation ensures
69 * mass conservation between the inlets of the blood vessel elements:
70 *
71 * \f[
72 * Q_{in}=\sum_{i}^{n_{outlets}} Q_{in, i}
73 * \f]
74 *
75 * ### Local contributions
76 *
77 * \f[
78 * \mathbf{y}^{e}=\left[\begin{array}{llllllllll}P_{in, 1}^{e} & Q_{in, 1}^{e}
79 * & P_{out, 1}^{e} & Q_{out, 1}^{e} &
80 * \dots & P_{out, i}^{e} & Q_{out, i}^{e} & Q_{in,1} & \dots &
81 * Q_{in,i}\end{array}\right] \f]
82 *
83 * Mass conservation
84 *
85 * \f[
86 * \mathbf{F}^{e} = \left[\begin{array}{llllllllll}
87 * 0 & 1 & 0 & 0 & \dots & 0 & 0 & -1 & \dots & -1 \\
88 * & & & & & & & & & \\
89 * \multicolumn{10}{c}{\mathrm{< blood \, vessel \, contributions >}} \\
90 * & & & & & & & & &
91 * \end{array}\right]
92 * \f]
93 *
94 * @tparam T Scalar type (e.g. `float`, `double`)
95 */

201template <typename T>
202void BloodVesselJunction<T>::setup_dofs(DOFHandler &dofhandler) {
203 // Set number of equations of a junction block based on number of
204 // inlets/outlets. Must be set before calling parent constructor
205 num_outlets = this->outlet_nodes.size();
206 std::list<std::string> internal_var_names;
207 for (size_t i = 0; i < num_outlets; i++) {
208 internal_var_names.push_back("flow_" + std::to_string(i));
209 }
210 Block<T>::setup_dofs_(dofhandler, 1, internal_var_names);
211 for (size_t i = 0; i < num_outlets; i++) {
212 blood_vessels.push_back(new BloodVessel<T>(
213 params.R[i], params.C[i], params.L[i], params.stenosis_coefficient[i],
214 this->name + "_bv" + std::to_string(i)));
215 blood_vessels[i]->inlet_nodes.push_back(this->inlet_nodes[0]);
216 blood_vessels[i]->outlet_nodes.push_back(this->outlet_nodes[i]);
217 blood_vessels[i]->setup_dofs(dofhandler);
218 blood_vessels[i]->global_var_ids[1] =
219 this->global_var_ids.end()[i - num_outlets];
220 }
221 num_triplets = {
222 {"F", 10 * num_outlets + num_outlets + 1},
223 {"E", 2 * num_outlets},
224 {"D", 2 * num_outlets},
225 };
226}

228template <typename T>
229void BloodVesselJunction<T>::update_constant(ALGEBRA::SparseSystem<T> &system) {
230 for (auto bv : blood_vessels) {
231 bv->update_constant(system);
232 }
233 // Mass conservation
234 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[1]) = 1.0;
235 for (size_t i = 0; i < num_outlets; i++) {
236 system.F.coeffRef(this->global_eqn_ids[0],
237 this->global_var_ids.end()[-(i + 1)]) = -1.0;
238 }
239}

241template <typename T>
242void BloodVesselJunction<T>::update_solution(
243 ALGEBRA::SparseSystem<T> &system, Eigen::Matrix<T, Eigen::Dynamic, 1> &y) {
244 for (auto bv : blood_vessels) {
245 bv->update_solution(system, y);
246 }
