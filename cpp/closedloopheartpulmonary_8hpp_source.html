45/**
46 * @brief Heart and pulmonary circulation model
47 *
48 * Models the mechanics of the 4 heart chambers and pulmonary circulation
49 *
50 * Reference for equations and model structure: Sankaran, S., Moghadam, M. E.,
51 * Kahn, A. M., Tseng, E. E., Guccione, J. M., & Marsden, A. L. (2012).
52 * Patient-specific multiscale modeling of blood flow for coronary artery bypass
53 * graft surgery. Annals of Biomedical Engineering, 40(10), 2228–2242.
54 * https://doi.org/10.1007/s10439-012-0579-3
55 *
56 * TODO: Equations and circuit diagram
57 *
58 * @tparam T Scalar type (e.g. `float`, `double`)
59 */
60template <typename T>
61class ClosedLoopHeartPulmonary : public Block<T> {
62 public:
63 /**
64 * @brief Parameters of the element.
65 *
66 * Struct containing all constant and/or time-dependent parameters of the
67 * element.
68 */
69 struct Parameters : public Block<T>::Parameters {
70 T Tsa; ///< Fractions of cardiac cycle (not sure)
71 T tpwave; ///< Fraction of cardiac cycle (P-wave)
72 T Erv_s; ///< Scaling for right ventricle elastance
73 T Elv_s; ///< Scaling for left ventricle elastance
74 T iml; ///< Scaling for intramyocardial pressure (left coronaries)
75 T imr; ///< Scaling for intramyocardial pressure (right coronaries)
76 T Lra_v; ///< Right atrium inductance
77 T Rra_v; ///< Right atrium outflow resistance
78 T Lrv_a; ///< Right ventricle inductance
79 T Rrv_a; ///< Right ventricle outflow resistance
80 T Lla_v; ///< Left atrium inductance
81 T Rla_v; ///< Left atrium outflow resistance
82 T Llv_a; ///< Left ventricle inductance
83 T Rlv_ao; ///< Left ventricle outflow resistance
84 T Vrv_u; ///< Right ventricle unstressed volume
85 T Vlv_u; ///< Right ventricle unstressed volume
86 T Rpd; ///< Pulmonary resistance
87 T Cp; ///< Pulmonary capacitance
88 T Cpa; ///< Aortic capacitance
89 T Kxp_ra; ///< Right atrium pressure-volume relationship (?)
90 T Kxv_ra; ///< Right atrium pressure-volume relationship (?)
91 T Kxp_la; ///< Left atrium pressure-volume relationship (?)
92 T Kxv_la; ///< Left atrium pressure-volume relationship (?)
93 T Emax_ra; ///< Right atrium elastance (?)
94 T Emax_la; ///< Left atrium elastance (?)
95 T Vaso_ra; ///< Right atrium rest volume (?)
96 T Vaso_la; ///< Left atrium rest volume (?)
97 };

356template <typename T>
357void ClosedLoopHeartPulmonary<T>::update_constant(
358 ALGEBRA::SparseSystem<T> &system) {
359 // DOF 2, Eq 1: Aortic pressure
360 system.E.coeffRef(this->global_eqn_ids[1], this->global_var_ids[2]) =
361 params.Cpa;
362 // DOF 4, Eq 2: Right atrium volume
363 system.E.coeffRef(this->global_eqn_ids[2], this->global_var_ids[4]) = 1.0;
364 // DOF 5, Eq 3: Right atrium outflow
365 system.E.coeffRef(this->global_eqn_ids[3], this->global_var_ids[5]) =
366 params.Lra_v;
367 // DOF 7, Eq 5: Right ventricle volume
368 system.E.coeffRef(this->global_eqn_ids[5], this->global_var_ids[7]) = 1.0;
369 // DOF 8, Eq 6: Right ventricle outflow
370 system.E.coeffRef(this->global_eqn_ids[6], this->global_var_ids[8]) =
371 params.Lrv_a;
372 // DOF 9, Eq 7: Pulmonary pressure
373 system.E.coeffRef(this->global_eqn_ids[7], this->global_var_ids[9]) =
374 params.Cp;
375 // DOF 11, Eq 9: Left atrium volume
376 system.E.coeffRef(this->global_eqn_ids[9], this->global_var_ids[11]) = 1.0;
377 // DOF 12, Eq 10: Left atrium outflow
378 system.E.coeffRef(this->global_eqn_ids[10], this->global_var_ids[12]) =
379 params.Lla_v;
380 // DOF 14, Eq 12: Left ventricle volume
381 system.E.coeffRef(this->global_eqn_ids[12], this->global_var_ids[14]) = 1.0;
382 // DOF 15, Eq 13: Left ventricle outflow
383 system.E.coeffRef(this->global_eqn_ids[13], this->global_var_ids[15]) =
384 params.Llv_a;
385}

393template <typename T>
394void ClosedLoopHeartPulmonary<T>::update_solution(
395 ALGEBRA::SparseSystem<T> &system, Eigen::Matrix<T, Eigen::Dynamic, 1> &y) {
396 this->get_psi_ra_la(y);
397 this->get_valve_positions(y);
398
399 // F and C matrices depend on time and solution
400 // Specifying all terms here, including constant terms (which can instead be
401 // specified in update_constant) for readability (Doesn't seem to make a
402 // difference to compute time) DOF IDs are arranged as inflow
403 // [P_in,Q_in,P_out,Q_out,internal variables...]
404
405 // DOF 0, Eq 0: Right atrium pressure
406 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[0]) = 1.0;
407 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[4]) =
408 -this->AA * params.Emax_ra;
409 system.C(this->global_eqn_ids[0]) =
410 this->AA * params.Emax_ra * params.Vaso_ra + psi_ra * (this->AA - 1.0);
411 system.D.coeffRef(this->global_eqn_ids[0], this->global_var_ids[4]) =
412 psi_ra_derivative * (this->AA - 1.0);
413
414 // DOF 1: Flow into right atrium (no equation)
415
416 // DOF 2, Eq 1: Aortic pressure
417 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[15]) =
418 -valves[15];
419 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[3]) = 1.0;
420
421 // DOF 3: Flow into aorta (no equation)
422
423 // DOF 4, Eq 2: Right atrium volume
424 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[5]) =
425 1.0 * valves[5];
426 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[1]) = -1.0;
427
428 // DOF 5, Eq 3: Right atrium outflow
429 system.F.coeffRef(this->global_eqn_ids[3], this->global_var_ids[5]) =
430 params.Rra_v * valves[5];
431 system.F.coeffRef(this->global_eqn_ids[3], this->global_var_ids[0]) = -1.0;
432 system.F.coeffRef(this->global_eqn_ids[3], this->global_var_ids[6]) = 1.0;
433
434 // DOF 6, Eq 4: Right ventricle pressure
435 system.F.coeffRef(this->global_eqn_ids[4], this->global_var_ids[6]) = 1.0;
436 system.F.coeffRef(this->global_eqn_ids[4], this->global_var_ids[7]) =
437 -this->Erv;
438 system.C(this->global_eqn_ids[4]) = this->Erv * params.Vrv_u;
439
440 // DOF 7, Eq 5: Right ventricle volume
441 system.F.coeffRef(this->global_eqn_ids[5], this->global_var_ids[5]) =
442 -1.0 * valves[5];
443 system.F.coeffRef(this->global_eqn_ids[5], this->global_var_ids[8]) =
444 1.0 * valves[8];
445
446 // DOF 8, Eq 6: Right ventricle outflow
447 system.F.coeffRef(this->global_eqn_ids[6], this->global_var_ids[6]) = -1.0;
448 system.F.coeffRef(this->global_eqn_ids[6], this->global_var_ids[9]) = 1.0;
449 system.F.coeffRef(this->global_eqn_ids[6], this->global_var_ids[8]) =
450 params.Rrv_a * valves[8];
451
452 // DOF 9, Eq 7: Pulmonary pressure
453 system.F.coeffRef(this->global_eqn_ids[7], this->global_var_ids[8]) =
454 -valves[8];
455 system.F.coeffRef(this->global_eqn_ids[7], this->global_var_ids[9]) =
456 1.0 / params.Rpd;
457 system.F.coeffRef(this->global_eqn_ids[7], this->global_var_ids[10]) =
458 -1.0 / params.Rpd;
459
460 // DOF 10, Eq 8: Left atrium pressure
461 system.F.coeffRef(this->global_eqn_ids[8], this->global_var_ids[10]) = 1.0;
462 system.F.coeffRef(this->global_eqn_ids[8], this->global_var_ids[11]) =
463 -this->AA * params.Emax_la;
464 system.C(this->global_eqn_ids[8]) =
465 this->AA * params.Emax_la * params.Vaso_la + psi_la * (this->AA - 1.0);
466 system.D.coeffRef(this->global_eqn_ids[8], this->global_var_ids[11]) =
467 psi_la_derivative * (this->AA - 1.0);
468
469 // DOF 11, Eq 9: Left atrium volume
470 system.F.coeffRef(this->global_eqn_ids[9], this->global_var_ids[8]) =
471 -1.0 * valves[8];
472 system.F.coeffRef(this->global_eqn_ids[9], this->global_var_ids[12]) =
473 1.0 * valves[12];
474
475 // DOF 12, Eq 10: Left atrium outflow
476 system.F.coeffRef(this->global_eqn_ids[10], this->global_var_ids[10]) = -1.0;
477 system.F.coeffRef(this->global_eqn_ids[10], this->global_var_ids[13]) = 1.0;
478 system.F.coeffRef(this->global_eqn_ids[10], this->global_var_ids[12]) =
479 params.Rla_v * valves[12];
480
481 // DOF 13, Eq 11: Left ventricle pressure
482 system.F.coeffRef(this->global_eqn_ids[11], this->global_var_ids[13]) = 1.0;
483 system.F.coeffRef(this->global_eqn_ids[11], this->global_var_ids[14]) =
484 -this->Elv;
485 system.C(this->global_eqn_ids[11]) = this->Elv * params.Vlv_u;
486
487 // DOF 14, Eq 12: Left ventricle volume
488 system.F.coeffRef(this->global_eqn_ids[12], this->global_var_ids[12]) =
489 -1.0 * valves[12];
490 system.F.coeffRef(this->global_eqn_ids[12], this->global_var_ids[15]) =
491 1.0 * valves[15];
492
493 // DOF 15, Eq 13: Left ventricle outflow
494 system.F.coeffRef(this->global_eqn_ids[13], this->global_var_ids[13]) = -1.0;
495 system.F.coeffRef(this->global_eqn_ids[13], this->global_var_ids[2]) = 1.0;
496 system.F.coeffRef(this->global_eqn_ids[13], this->global_var_ids[15]) =
497 params.Rlv_ao * valves[15];
498}

500template <typename T>
501void ClosedLoopHeartPulmonary<T>::get_activation_and_elastance_functions(
502 T time) {
503 T T_cardiac = this->cardiac_cycle_period;
504 T Tsa = T_cardiac * params.Tsa;
505 T tpwave = T_cardiac / params.tpwave;
506 T t_in_cycle = fmod(time, T_cardiac);
507
508 // Activation function
509 AA = 0.0;
510 if (t_in_cycle <= tpwave) {
511 AA = (0.5) * (1.0 - cos(2.0 * PI * (t_in_cycle - tpwave + Tsa) / Tsa));
512 } else if ((t_in_cycle >= (T_cardiac - Tsa) + tpwave) and
513 (t_in_cycle < T_cardiac)) {
514 AA =
515 (0.5) *
516 (1.0 - cos(2.0 * PI * (t_in_cycle - tpwave - (T_cardiac - Tsa)) / Tsa));
517 } else {
518 AA = 0.0;
519 }
520
521 // Elastance modes (copied from J. Tran's tuning framework)
522 const int num_elast_modes = 25;
523 T Ft_elastance[num_elast_modes][2] = {
524 {0.283748803, 0.000000000}, {0.031830626, -0.374299825},
525 {-0.209472400, -0.018127770}, {0.020520047, 0.073971113},
526 {0.008316883, -0.047249597}, {-0.041677660, 0.003212163},
527 {0.000867323, 0.019441411}, {-0.001675379, -0.005565534},
528 {-0.011252277, 0.003401432}, {-0.000414677, 0.008376795},
529 {0.000253749, -0.000071880}, {-0.002584966, 0.001566861},
530 {0.000584752, 0.003143555}, {0.000028502, -0.000024787},
531 {0.000022961, -0.000007476}, {0.000018735, -0.000001281},
532 {0.000015573, 0.000001781}, {0.000013133, 0.000003494},
533 {0.000011199, 0.000004507}, {0.000009634, 0.000005117},
534 {0.000008343, 0.000005481}, {0.000007265, 0.000005687},
535 {0.000006354, 0.000005789}, {0.000005575, 0.000005821},
536 {0.000004903, 0.000005805}};
537
538 // RV and LV elastance
539 T Elv_i = 0.0;
540 for (auto i = 0; i < num_elast_modes; i++)
541 Elv_i = Elv_i +
542 (Ft_elastance[i][0]) * cos(2.0 * PI * i * t_in_cycle / T_cardiac) -
543 (Ft_elastance[i][1]) * sin(2.0 * PI * i * t_in_cycle / T_cardiac);
544
545 Elv = Elv_i * params.Elv_s;
546 Erv = Elv_i * params.Erv_s;
547}

549template <typename T>
550void ClosedLoopHeartPulmonary<T>::get_psi_ra_la(
551 Eigen::Matrix<T, Eigen::Dynamic, 1> &y) {
552 auto RA_volume = y[this->global_var_ids[4]];
553 auto LA_volume = y[this->global_var_ids[11]];
554 psi_ra =
555 params.Kxp_ra * (exp((RA_volume - params.Vaso_ra) * params.Kxv_ra) - 1.0);
556 psi_la =
557 params.Kxp_la * (exp((LA_volume - params.Vaso_la) * params.Kxv_la) - 1.0);
558
559 psi_ra_derivative = params.Kxp_ra *
560 exp((RA_volume - params.Vaso_ra) * params.Kxv_ra) *
561 params.Kxv_ra;
562 psi_la_derivative = params.Kxp_la *
563 exp((LA_volume - params.Vaso_la) * params.Kxv_la) *
564 params.Kxv_la;
565}

567template <typename T>
568void ClosedLoopHeartPulmonary<T>::get_valve_positions(
569 Eigen::Matrix<T, Eigen::Dynamic, 1> &y) {
570 std::fill(valves, valves + 16, 1.0);
571
572 // RA to RV
573 auto pressure_ra = y[this->global_var_ids[0]];
574 auto pressure_rv = y[this->global_var_ids[6]];
575 auto outflow_ra = y[this->global_var_ids[5]];
576 if ((pressure_ra <= pressure_rv) and (outflow_ra <= 0.0)) {
577 valves[5] = 0.0;
578 y[this->global_var_ids[5]] = 0.0;
579 }
580
581 // RV to pulmonary
582 auto pressure_pulmonary = y[this->global_var_ids[9]];
583 auto outflow_rv = y[this->global_var_ids[8]];
584 if ((pressure_rv <= pressure_pulmonary) and (outflow_rv <= 0.0)) {
585 valves[8] = 0.0;
586 y[this->global_var_ids[8]] = 0.0;
587 }
588
589 // LA to LV
590 auto pressure_la = y[this->global_var_ids[10]];
591 auto pressure_lv = y[this->global_var_ids[13]];
592 auto outflow_la = y[this->global_var_ids[12]];
593 if ((pressure_la <= pressure_lv) and (outflow_la <= 0.0)) {
594 valves[12] = 0.0;
595 y[this->global_var_ids[12]] = 0.0;
596 }
597
598 // LV to aorta
599 auto pressure_aorta = y[this->global_var_ids[2]];
600 auto outflow_lv = y[this->global_var_ids[15]];
601 if ((pressure_lv <= pressure_aorta) and (outflow_lv <= 0.0)) {
602 valves[15] = 0.0;
603 y[this->global_var_ids[15]] = 0.0;
604 }
605}
