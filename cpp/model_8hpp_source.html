171template <typename T>
172std::map<std::string, int> Model<T>::get_num_triplets() {
173 std::map<std::string, int> num_triplets = {
174 {"F", 0},
175 {"E", 0},
176 {"D", 0},
177 };
178 for (auto &elem : blocks) {
179 for (auto &[key, value] : elem->get_num_triplets()) {
180 num_triplets[key] += value;
181 }
182 }
183 return num_triplets;
184}
