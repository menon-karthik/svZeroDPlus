114template <typename T>
115void ConfigReader<T>::load(std::string &specifier) {
116 // Create iterator for json configuration
117 simdjson::ondemand::parser parser;
118 simdjson::padded_string string;
119 if (HELPERS::startswith(specifier, "{")) {
120 string = simdjson::padded_string(specifier);
121 } else {
122 string = simdjson::padded_string::load(specifier);
123 }
124 auto config = parser.iterate(string);
125
126 // Load simulation parameters
127 auto sim_params = config["simulation_parameters"];
128 sim_num_cycles = sim_params["number_of_cardiac_cycles"].get_int64();
129 sim_pts_per_cycle =
130 sim_params["number_of_time_pts_per_cardiac_cycle"].get_int64();
131 sim_num_time_steps = (sim_pts_per_cycle - 1.0) * sim_num_cycles + 1.0;
132 sim_cardiac_cycle_period = -1.0; // Negative value indicates this has not
133 // been read from config file yet.
134 try {
135 sim_abs_tol = sim_params["absolute_tolerance"].get_double();
136 } catch (simdjson::simdjson_error) {
137 sim_abs_tol = 1e-8;
138 }
139 try {
140 sim_nliter = sim_params["maximum_nonlinear_iterations"].get_int64();
141 } catch (simdjson::simdjson_error) {
142 sim_nliter = 30;
143 }
144 try {
145 sim_steady_initial = sim_params["steady_initial"].get_bool();
146 } catch (simdjson::simdjson_error) {
147 sim_steady_initial = true;
148 }
149 try {
150 output_variable_based = sim_params["output_variable_based"].get_bool();
151 } catch (simdjson::simdjson_error) {
152 output_variable_based = false;
153 }
154 try {
155 output_interval = sim_params["output_interval"].get_int64();
156 } catch (simdjson::simdjson_error) {
157 output_interval = 1;
158 }
159 try {
160 output_mean_only = sim_params["output_mean_only"].get_bool();
161 } catch (simdjson::simdjson_error) {
162 output_mean_only = false;
163 }
164 try {
165 output_derivative = sim_params["output_derivative"].get_bool();
166 } catch (simdjson::simdjson_error) {
167 output_derivative = false;
168 }
169 try {
170 output_last_cycle_only = sim_params["output_last_cycle_only"].get_bool();
171 } catch (simdjson::simdjson_error) {
172 output_last_cycle_only = false;
173 }
174
175 // Create list to store block connections while generating blocks
176 std::vector<std::tuple<std::string_view, std::string_view>> connections;
177
178 // Create vessels
179 std::map<std::int64_t, std::string_view> vessel_id_map;
180 for (auto vessel_config : config["vessels"]) {
181 auto vessel_values = vessel_config["zero_d_element_values"];
182 std::string_view vessel_name = vessel_config["vessel_name"].get_string();
183 vessel_id_map.insert({vessel_config["vessel_id"].get_int64(), vessel_name});
184 if (std::string_view(vessel_config["zero_d_element_type"].get_string()) ==
185 "BloodVessel") {
186 T R = vessel_values["R_poiseuille"].get_double();
187 T C;
188 try {
189 C = vessel_values["C"].get_double();
190 } catch (simdjson::simdjson_error) {
191 C = 0.0;
192 }
193 T L;
194 try {
195 L = vessel_values["L"].get_double();
196 } catch (simdjson::simdjson_error) {
197 L = 0.0;
198 }
199 T stenosis_coefficient;
200 try {
201 stenosis_coefficient =
202 vessel_values["stenosis_coefficient"].get_double();
203 } catch (simdjson::simdjson_error) {
204 stenosis_coefficient = 0.0;
205 }
206 model.blocks.push_back(new MODEL::BloodVessel<T>(
207 R = R, C = C, L = L, stenosis_coefficient = stenosis_coefficient,
208 static_cast<std::string>(vessel_name)));
209 DEBUG_MSG("Created vessel " << vessel_name);
210 } else {
211 throw std::invalid_argument("Unknown vessel type");
212 }
213
214 // Read connected boundary conditions
215 try {
216 auto inlet_bc = vessel_config["boundary_conditions"]["inlet"];
217 connections.push_back({inlet_bc.get_string(), vessel_name});
218 } catch (simdjson::simdjson_error) {
219 }
220
221 try {
222 auto outlet_bc = vessel_config["boundary_conditions"]["outlet"];
223 connections.push_back({vessel_name, outlet_bc.get_string()});
224 } catch (simdjson::simdjson_error) {
225 }
226 }
227
228 std::vector<std::string_view> closed_loop_bcs;
229
230 // Create boundary conditions
231 for (auto bc_config : config["boundary_conditions"]) {
232 std::string_view bc_type = bc_config["bc_type"];
233 std::string_view bc_name = bc_config["bc_name"];
234 auto bc_values = bc_config["bc_values"];
235
236 std::vector<T> t;
237 try {
238 for (auto x : bc_values["t"]) {
239 t.push_back(x.get_double());
240 }
241 } catch (simdjson::simdjson_error) {
242 t.push_back(0.0);
243 };
244
245 if (bc_type == "RCR") {
246 T Rp = bc_values["Rp"], C = bc_values["C"], Rd = bc_values["Rd"],
247 Pd = bc_values["Pd"];
248 model.blocks.push_back(new MODEL::WindkesselBC<T>(
249 Rp = Rp, C = C, Rd = Rd, Pd = Pd, static_cast<std::string>(bc_name)));
250 DEBUG_MSG("Created boundary condition " << bc_name);
251 } else if (bc_type == "ClosedLoopRCR") {
252 T Rp = bc_values["Rp"], C = bc_values["C"], Rd = bc_values["Rd"];
253 bool closed_loop_outlet = bc_values["closed_loop_outlet"];
254 if (closed_loop_outlet == true) {
255 closed_loop_bcs.push_back(bc_name);
256 }
257 model.blocks.push_back(new MODEL::ClosedLoopRCRBC<T>(
258 Rp = Rp, C = C, Rd = Rd, closed_loop_outlet = closed_loop_outlet,
259 static_cast<std::string>(bc_name)));
260 DEBUG_MSG("Created boundary condition " << bc_name);
261 } else if (bc_type == "FLOW") {
262 std::vector<T> Q;
263 try {
264 for (auto x : bc_values["Q"].get_array()) {
265 Q.push_back(x.get_double());
266 }
267 } catch (simdjson::simdjson_error) {
268 Q.push_back(bc_values["Q"].get_double());
269 }
270
271 MODEL::TimeDependentParameter q_param(t, Q);
272 if (q_param.isconstant == false) {
273 if ((sim_cardiac_cycle_period > 0.0) &&
274 (q_param.cycle_period != sim_cardiac_cycle_period)) {
275 throw std::runtime_error(
276 "Inconsistent cardiac cycle period defined in "
277 "TimeDependentParameter");
278 } else {
279 sim_cardiac_cycle_period = q_param.cycle_period;
280 }
281 }
282 model.blocks.push_back(new MODEL::FlowReferenceBC<T>(
283 q_param, static_cast<std::string>(bc_name)));
284 DEBUG_MSG("Created boundary condition " << bc_name);
285
286 } else if (bc_type == "RESISTANCE") {
287 std::vector<T> R;
288 try {
289 for (auto x : bc_values["R"].get_array()) {
290 R.push_back(x.get_double());
291 }
292 } catch (simdjson::simdjson_error) {
293 R.push_back(bc_values["R"].get_double());
294 }
295 MODEL::TimeDependentParameter r_param(t, R);
296
297 std::vector<T> Pd;
298 try {
299 for (auto x : bc_values["Pd"].get_array()) {
300 Pd.push_back(x.get_double());
301 }
302 } catch (simdjson::simdjson_error) {
303 Pd.push_back(bc_values["Pd"].get_double());
304 }
305 MODEL::TimeDependentParameter pd_param(t, Pd);
306 model.blocks.push_back(new MODEL::ResistanceBC<T>(
307 r_param, pd_param, static_cast<std::string>(bc_name)));
308 DEBUG_MSG("Created boundary condition " << bc_name);
309 } else if (bc_type == "PRESSURE") {
310 std::vector<T> P;
311 try {
312 for (auto x : bc_values["P"].get_array()) {
313 P.push_back(x.get_double());
314 }
315 } catch (simdjson::simdjson_error) {
316 P.push_back(bc_values["P"].get_double());
317 }
318 MODEL::TimeDependentParameter p_param(t, P);
319 if (p_param.isconstant == false) {
320 if ((sim_cardiac_cycle_period > 0.0) &&
321 (p_param.cycle_period != sim_cardiac_cycle_period)) {
322 throw std::runtime_error(
323 "Inconsistent cardiac cycle period defined in "
324 "TimeDependentParameter");
325 } else {
326 sim_cardiac_cycle_period = p_param.cycle_period;
327 }
328 }
329 model.blocks.push_back(new MODEL::PressureReferenceBC<T>(
330 p_param, static_cast<std::string>(bc_name)));
331 DEBUG_MSG("Created boundary condition " << bc_name);
332 } else if (bc_type == "CORONARY") {
333 T Ra = bc_values["Ra1"], Ram = bc_values["Ra2"], Rv = bc_values["Rv1"],
334 Ca = bc_values["Ca"], Cim = bc_values["Cc"];
335 auto Pim_json = bc_values["Pim"], Pv_json = bc_values["P_v"];
336
337 std::vector<T> Pim;
338 try {
339 for (auto x : bc_values["Pim"].get_array()) {
340 Pim.push_back(x.get_double());
341 }
342 } catch (simdjson::simdjson_error) {
343 Pim.push_back(bc_values["Pim"].get_double());
344 }
345 MODEL::TimeDependentParameter pim_param(t, Pim);
346 std::vector<T> P_v;
347 try {
348 for (auto x : bc_values["P_v"].get_array()) {
349 P_v.push_back(x.get_double());
350 }
351 } catch (simdjson::simdjson_error) {
352 P_v.push_back(bc_values["P_v"].get_double());
353 }
354 MODEL::TimeDependentParameter pv_param(t, P_v);
355
356 model.blocks.push_back(new MODEL::OpenLoopCoronaryBC<T>(
357 Ra = Ra, Ram = Ram, Rv = Rv, Ca = Ca, Cim = Cim, pim_param, pv_param,
358 static_cast<std::string>(bc_name)));
359 DEBUG_MSG("Created boundary condition " << bc_name);
360 } else if (bc_type == "ClosedLoopCoronary") {
361 T Ra = bc_values["Ra"], Ram = bc_values["Ram"], Rv = bc_values["Rv"],
362 Ca = bc_values["Ca"], Cim = bc_values["Cim"];
363 std::string_view side = bc_values["side"];
364 closed_loop_bcs.push_back(bc_name);
365 // std::string side = static_cast<std::string>(bc_values["side"]);
366 model.blocks.push_back(new MODEL::ClosedLoopCoronaryBC<T>(
367 Ra = Ra, Ram = Ram, Rv = Rv, Ca = Ca, Cim = Cim,
368 static_cast<std::string>(side), static_cast<std::string>(bc_name)));
369 DEBUG_MSG("Created boundary condition " << bc_name);
370 } else {
371 throw std::invalid_argument("Unknown boundary condition type");
372 }
373 }
374
375 // Create junctions
376 for (auto junction_config : config["junctions"]) {
377 std::string_view j_type = junction_config["junction_type"].get_string();
378 std::string_view junction_name =
379 junction_config["junction_name"].get_string();
380 if ((j_type == "NORMAL_JUNCTION") || (j_type == "internal_junction")) {
381 model.blocks.push_back(
382 new MODEL::Junction<T>(static_cast<std::string>(junction_name)));
383 } else if (j_type == "resistive_junction") {
384 std::vector<T> R;
385 for (auto x : junction_config["junction_values"]["R"].get_array()) {
386 R.push_back(x.get_double());
387 }
388 model.blocks.push_back(new MODEL::ResistiveJunction<T>(
389 R, static_cast<std::string>(junction_name)));
390 } else if (j_type == "BloodVesselJunction") {
391 std::vector<T> R;
392 std::vector<T> C;
393 std::vector<T> L;
394 std::vector<T> stenosis_coefficient;
395 for (auto x :
396 junction_config["junction_values"]["R_poiseuille"].get_array()) {
397 R.push_back(x.get_double());
398 }
399 for (auto x : junction_config["junction_values"]["C"].get_array()) {
400 C.push_back(x.get_double());
401 }
402 for (auto x : junction_config["junction_values"]["L"].get_array()) {
403 L.push_back(x.get_double());
404 }
405 for (auto x : junction_config["junction_values"]["stenosis_coefficient"]
406 .get_array()) {
407 stenosis_coefficient.push_back(x.get_double());
408 }
409 model.blocks.push_back(new MODEL::BloodVesselJunction<T>(
410 R, C, L, stenosis_coefficient,
411 static_cast<std::string>(junction_name)));
412 } else {
413 throw std::invalid_argument("Unknown junction type");
414 }
415 // Check for connections to inlet and outlet vessels and append to
416 // connections list
417 for (auto inlet_vessel : junction_config["inlet_vessels"]) {
418 connections.push_back(
419 {vessel_id_map[inlet_vessel.get_int64()], junction_name});
420 }
421 for (auto outlet_vessel : junction_config["outlet_vessels"]) {
422 connections.push_back(
423 {junction_name, vessel_id_map[outlet_vessel.get_int64()]});
424 }
425 DEBUG_MSG("Created junction " << junction_name);
426 }
427
428 // Create closed-loop blocks
429 bool heartpulmonary_block_present =
430 false; ///< Flag to check if heart block is present (requires different
431 ///< handling)
432 try {
433 for (auto closed_loop_config : config["closed_loop_blocks"]) {
434 std::string_view closed_loop_type =
435 closed_loop_config["closed_loop_type"];
436 if (closed_loop_type == "ClosedLoopHeartAndPulmonary") {
437 if (heartpulmonary_block_present == false) {
438 heartpulmonary_block_present = true;
439 sim_steady_initial = false;
440 std::string_view heartpulmonary_name = "CLH";
441 T cycle_period =
442 closed_loop_config["cardiac_cycle_period"].get_double();
443 if ((sim_cardiac_cycle_period > 0.0) &&
444 (cycle_period != sim_cardiac_cycle_period)) {
445 throw std::runtime_error(
446 "Inconsistent cardiac cycle period defined in "
447 "ClosedLoopHeartAndPulmonary.");
448 } else {
449 sim_cardiac_cycle_period = cycle_period;
450 }
451 auto heart_params = closed_loop_config["parameters"];
452 // Convert to std::map to keep model blocks independent of simdjson
453 std::map<std::string, T> param_values;
454 param_values.insert(std::make_pair("Tsa", heart_params["Tsa"]));
455 param_values.insert(std::make_pair("tpwave", heart_params["tpwave"]));
456 param_values.insert(std::make_pair("Erv_s", heart_params["Erv_s"]));
457 param_values.insert(std::make_pair("Elv_s", heart_params["Elv_s"]));
458 param_values.insert(std::make_pair("iml", heart_params["iml"]));
459 param_values.insert(std::make_pair("imr", heart_params["imr"]));
460 param_values.insert(std::make_pair("Lra_v", heart_params["Lra_v"]));
461 param_values.insert(std::make_pair("Rra_v", heart_params["Rra_v"]));
462 param_values.insert(std::make_pair("Lrv_a", heart_params["Lrv_a"]));
463 param_values.insert(std::make_pair("Rrv_a", heart_params["Rrv_a"]));
464 param_values.insert(std::make_pair("Lla_v", heart_params["Lla_v"]));
465 param_values.insert(std::make_pair("Rla_v", heart_params["Rla_v"]));
466 param_values.insert(std::make_pair("Llv_a", heart_params["Llv_a"]));
467 param_values.insert(std::make_pair("Rlv_ao", heart_params["Rlv_ao"]));
468 param_values.insert(std::make_pair("Vrv_u", heart_params["Vrv_u"]));
469 param_values.insert(std::make_pair("Vlv_u", heart_params["Vlv_u"]));
470 param_values.insert(std::make_pair("Rpd", heart_params["Rpd"]));
471 param_values.insert(std::make_pair("Cp", heart_params["Cp"]));
472 param_values.insert(std::make_pair("Cpa", heart_params["Cpa"]));
473 param_values.insert(std::make_pair("Kxp_ra", heart_params["Kxp_ra"]));
474 param_values.insert(std::make_pair("Kxv_ra", heart_params["Kxv_ra"]));
475 param_values.insert(std::make_pair("Kxp_la", heart_params["Kxp_la"]));
476 param_values.insert(std::make_pair("Kxv_la", heart_params["Kxv_la"]));
477 param_values.insert(
478 std::make_pair("Emax_ra", heart_params["Emax_ra"]));
479 param_values.insert(
480 std::make_pair("Emax_la", heart_params["Emax_la"]));
481 param_values.insert(
482 std::make_pair("Vaso_ra", heart_params["Vaso_ra"]));
483 param_values.insert(
484 std::make_pair("Vaso_la", heart_params["Vaso_la"]));
485 if (param_values.size() == 27) {
486 model.blocks.push_back(new MODEL::ClosedLoopHeartPulmonary<T>(
487 param_values, cycle_period,
488 static_cast<std::string>(heartpulmonary_name)));
489 } else {
490 throw std::runtime_error(
491 "Error. ClosedLoopHeartAndPulmonary should have 27 parameters");
492 }
493 // Junction at inlet to heart
494 std::string_view heart_inlet_junction_name = "J_heart_inlet";
495 connections.push_back(
496 {heart_inlet_junction_name, heartpulmonary_name});
497 model.blocks.push_back(new MODEL::Junction<T>(
498 static_cast<std::string>(heart_inlet_junction_name)));
499 for (auto heart_inlet_elem : closed_loop_bcs) {
500 connections.push_back(
501 {heart_inlet_elem, heart_inlet_junction_name});
502 }
503 // Junction at outlet from heart
504 std::string_view heart_outlet_junction_name = "J_heart_outlet";
505 connections.push_back(
506 {heartpulmonary_name, heart_outlet_junction_name});
507 model.blocks.push_back(new MODEL::Junction<T>(
508 static_cast<std::string>(heart_outlet_junction_name)));
509 for (auto heart_outlet_block : closed_loop_config["outlet_blocks"]) {
510 connections.push_back(
511 {heart_outlet_junction_name, heart_outlet_block});
512 }
513 } else {
514 throw std::runtime_error(
515 "Error. Only one ClosedLoopHeartAndPulmonary can be included.");
516 }
517 }
518 }
519 } catch (simdjson::simdjson_error) {
520 }
521
522 // Create Connections
523 for (auto &connection : connections) {
524 for (auto &ele1 : model.blocks) {
525 for (auto &ele2 : model.blocks) {
526 if ((ele1->name == std::get<0>(connection)) &&
527 (ele2->name == std::get<1>(connection))) {
528 model.nodes.push_back(new MODEL::Node(ele1->name + ":" + ele2->name));
529 DEBUG_MSG("Created node " << model.nodes.back()->name);
530 ele1->outlet_nodes.push_back(model.nodes.back());
531 ele2->inlet_nodes.push_back(model.nodes.back());
532 model.nodes.back()->setup_dofs(model.dofhandler);
533 }
534 }
535 }
536 }
537
538 // Setup degrees of freedom of the system
539 for (auto &block : model.blocks) {
540 block->setup_dofs(model.dofhandler);
541 }
542
543 // Set value of cardiac cycle period
544 if (sim_cardiac_cycle_period < 0.0) {
545 sim_cardiac_cycle_period =
546 1.0; // If it has not been read from config or TimeDependentParameter
547 // yet, set as default value of 1.0
548 }
549 // Calculate time step size
550 sim_time_step_size = sim_cardiac_cycle_period / (T(sim_pts_per_cycle) - 1.0);
551
552 // Update block parameters that depend on DOFs and other params of other
553 // blocks For example, coronary block params that depend on heart block DOFs
554 for (auto &block : model.blocks) {
555 block->update_model_dependent_params(model);
556 }
557
558 // Read initial condition
559 initial_state = ALGEBRA::State<T>::Zero(model.dofhandler.size());
560 // Initialize blocks that have fixed initial conditions
561 for (auto &block : model.blocks) {
562 block->set_ICs(initial_state);
563 }
564 try {
565 auto initial_condition = config["initial_condition"].value();
566 for (size_t i = 0; i < model.dofhandler.size(); i++) {
567 try {
568 initial_state.y[i] = initial_condition[model.dofhandler.variables[i]];
569 } catch (simdjson::simdjson_error) {
570 DEBUG_MSG("Initial condition found, but no value for "
571 << model.dofhandler.variables[i]);
572 }
573 }
574 } catch (simdjson::simdjson_error) {
575 }
576}
