\doxysubsection{Detailed Description}
\subsubsection*{template$<$typename T$>$\newline
class MODEL\+::\+Blood\+Vessel$<$ T $>$}
Resistor-\/capacitor-\/inductor blood vessel with optional stenosis. 

Models the mechanical behavior of a bloodvessel with optional stenosis.

\[
\begin{circuitikz} \draw
node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
\draw (1,0) node[anchor=south]{$P_{in}$}
to [R, l=$R$, *-] (3,0)
to [R, l=$R_{ste}$, -] (5,0) node[anchor=south] {$P_{C}$}
(5,0) to [L, l=$L$, *-*] (7,0)
node[anchor=south]{$P_{out}$}
(5,0) to [C, l=$C$, *-] (5,-1.5)
node[ground]{};
\draw [-latex] (7.2,0) -- (8,0) node[right] {$Q_{out}$};
\end{circuitikz}
\]\hypertarget{class_m_o_d_e_l_1_1_blood_vessel_autotoc_md15}{}\doxyparagraph{Governing equations}\label{class_m_o_d_e_l_1_1_blood_vessel_autotoc_md15}
\[
P_{in}^{e}-P_{out}^{e}-(R+R_{ste}) Q_{in}^{e}-L\frac{d Q_{out}^{e}}{dt}=0
\]

\[
Q_{i n}^{e}-Q_{o u t}^{e}-C \frac{d P_{c}^{e}}{d t}=0
\]

\[
P_{i n}^{e}-(R+R_{ste}) Q_{i n}^{e}-P_{c}=0
\]\hypertarget{class_m_o_d_e_l_1_1_blood_vessel_autotoc_md16}{}\doxyparagraph{Local contributions}\label{class_m_o_d_e_l_1_1_blood_vessel_autotoc_md16}
\[
\mathbf{y}^{e}=\left[\begin{array}{lllll}P_{i n}^{e} & Q_{in}^{e} &
P_{out}^{e} & Q_{out}^{e} & P_C\end{array}\right]^{T} \]

\[
\mathbf{E}^{e}=\left[\begin{array}{ccccc}
0 & 0 & 0 & -L & 0 \\
0 & 0 & 0 & 0 & -C \\
0 & 0 & 0 & 0 & 0
\end{array}\right]
\]

\[
\mathbf{F}^{e}=\left[\begin{array}{ccccc}
1 & -R_{ste}-R & -1 & 0 & 0 \\
0 & 1 & 0 & -1 & 0 \\
1 & -R_{ste}-R & 0 & 0 & -1
\end{array}\right]
\]

\[
\mathbf{D}^{e}=\left[\begin{array}{ccccc}
0 & -R_{ste} & 0 & 0 & 0 \\
0 & 0 & 0 & 0 & 0 \\
0 & -R_{ste} & 0 & 0 & 0
\end{array}\right]
\]

with the stenosis resistance $ R_{ste}=K_{t} \frac{\rho}{2
A_{o}^{2}}\left(\frac{A_{o}}{A_{s}}-1\right)^{2}|Q_{in}^{e}| $. The constant part of the equation is summarized in \mbox{\hyperlink{struct_m_o_d_e_l_1_1_blood_vessel_1_1_parameters_a5f0adc244080b0bd404c579d39ec3f2b}{Parameters\+::stenosis\+\_\+coefficient}}. $R$, $C$, and $L$ refer to Poisieuille resistance, capacitance and inductance, respectively.
